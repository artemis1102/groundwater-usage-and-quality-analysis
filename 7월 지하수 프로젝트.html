import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

#한글 폰트 설정 (로컬 환경에서만 적용, 없으면 맑은고딕 또는 나눔고딕 설치 필요)
plt.rcParams['font.family'] = 'Malgun Gothic'  # 윈도우용
plt.rcParams['axes.unicode_minus'] = False

#엑셀 파일 불러오기 (header=1로 실제 데이터 시작 지점 설정)
df_생활용 = pd.read_excel("세부용도별_2023년_서울특별시_생활용 지하수.xlsx", header=1)
df_농업용 = pd.read_excel("세부용도별_2023년_서울특별시_농·어업용 지하수.xlsx", header=1)
df_공업용 = pd.read_excel("세부용도별_2023년_서울특별시_공업용 지하수.xlsx", header=1)

#필요한 열만 선택해서 정리
df_생활용 = df_생활용[['행정구역', '총계']]
df_농업용 = df_농업용[['행정구역', '총계']]
df_공업용 = df_공업용[['행정구역', '총계']]

#열 이름 통일
df_생활용.columns = ['행정구역', '생활용']
df_농업용.columns = ['행정구역', '농업용']
df_공업용.columns = ['행정구역', '공업용']

#데이터 병합 (공통된 행정구역 기준으로 merge)
df_merged = df_생활용.merge(df_농업용, on='행정구역').merge(df_공업용, on='행정구역')

#숫자형으로 변환 (오류 발생 시 NaN으로 처리)
for col in ['생활용', '농업용', '공업용']:
    df_merged[col] = pd.to_numeric(df_merged[col], errors='coerce')

# 각화를 위한 데이터 재구성 (melt)
df_melted = df_merged.melt(id_vars='행정구역', var_name='용도', value_name='사용량(톤/년)')

#막대 그래프 시각화
plt.figure(figsize=(14, 7))
sns.barplot(data=df_melted, x='행정구역', y='사용량(톤/년)', hue='용도')
plt.title('2023년 서울시 구별 지하수 사용량 (생활용, 농업용, 공업용)')
plt.xticks(rotation=45)
plt.tight_layout()
plt.show()
