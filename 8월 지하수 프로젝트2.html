# 오염여부와 위배항목 판별 함수
def evaluate_row(row):
    violations = []
    for col, rule in criteria.items():
        val = row[col]
        if pd.isna(val):  # 결측은 판단 보류
            continue
        if len(rule) == 2:
            low, high = rule
            if (low is not None and val < low) or (high is not None and val > high):
                violations.append(col)
        elif len(rule) == 1:  # 불검출 조건
            if val > 0:
                violations.append(col)
    return 1 if violations else 0, ",".join(violations)

# 판별 적용
df[["오염여부", "위배항목"]] = df.apply(lambda row: pd.Series(evaluate_row(row)), axis=1)

    '트리클로로에틸렌_TCE': '휘발성 유기염소계',

cause_order = [
    'pH 이상(산성/알칼리)', '유기물 오염', '비료/질소성분', '염류(염분) 축적',
    '중금속', '유해 독성물질', '농약/유기인계', '유기화합물(페놀)', '휘발성 유기염소계'
]

# 3) 수치형 변환
for col in criteria:
    new_df[col] = pd.to_numeric(new_df[col], errors='coerce')

# 4) 평가 함수
def evaluate_row(row):
    violations = []
    causes = set()
    for col, rule in criteria.items():
        val = row[col]
        if pd.isna(val):
            continue
        violated = False
        if len(rule) == 2:
            low, high = rule
            if (low is not None and val < low) or (high is not None and val > high):
                violated = True
        else:  # len(rule) == 1
            if val > 0:
                violated = True
        if violated:
            violations.append(col)
            causes.add(cause_map.get(col, '기타'))
    polluted = 1 if violations else 0
    if causes:
        ordered = sorted(causes, key=lambda c: cause_order.index(c) if c in cause_order else 999)
        cause_str = ",".join(ordered)
    else:
        cause_str = np.nan
    return pd.Series({'오염여부': polluted,
                      '위배항목': ",".join(violations),
                      '오염원인': cause_str})

# 5) 적용
new_df[['오염여부', '위배항목', '오염원인']] = new_df.apply(evaluate_row, axis=1)

# 2. 사용할 주요 열만 선택 (수질 항목 + 지역 변수)
selected_cols = [
    '수소이온농도_pH', '화학적산소요구량_COD', '질산성질소_NO3_N', '염소이온_Cl_',
    '카드뮴_Cd', '비소_As', '시안_CN', '수은_Hg', '유기인', '페놀_Phenol',
    '납_Pb', '6가크롬_Cr+6', '트리클로로에틸렌_TCE', '테트라클로로에틸렌',
    '시도', '수문지질', '오염여부'  # target 포함
]
df2 = df[selected_cols]

# 3. 결측값 제거 또는 평균 대체 (빠르게 처리)
df2 = df2.dropna(subset=["오염여부"])
df2.fillna(df2.mean(numeric_only=True), inplace=True)

df2 = pd.get_dummies(df2, columns=["시도", "수문지질"], drop_first=True)

X = df2.drop(columns=["오염여부"])

# 6. 모델 학습 및 평가 (적합수가 90%이상이므로 데이터 불균형. balanced옵션 활용)

y_pred = model.predict(X_test)
