# -*- coding: utf-8 -*-
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# -----------------------------
# 0) 한글 폰트 (로컬 환경에 맞게)
# -----------------------------
plt.rcParams['font.family'] = 'Malgun Gothic'  # 윈도우: 맑은 고딕
plt.rcParams['axes.unicode_minus'] = False

# -----------------------------
# 1) 엑셀 불러오기 (header=1)
#    파일명은 현재 폴더 기준 (경로 다르면 바꿔주세요)
# -----------------------------
# 2021
df21_life = pd.read_excel("세부용도별_2021년_서울특별시_생활용 지하수.xlsx", header=1)
df21_agri = pd.read_excel("세부용도별_2021년_서울특별시_농·어업용 지하수.xlsx", header=1)
df21_ind  = pd.read_excel("세부용도별_2021년_서울특별시_공업용 지하수.xlsx", header=1)

# 2022
df22_life = pd.read_excel("세부용도별_2022년_서울특별시_생활용 지하수.xlsx", header=1)
df22_agri = pd.read_excel("세부용도별_2022년_서울특별시_농·어업용 지하수.xlsx", header=1)
df22_ind  = pd.read_excel("세부용도별_2022년_서울특별시_공업용 지하수.xlsx", header=1)

# 2023
df23_life = pd.read_excel("세부용도별_2023년_서울특별시_생활용 지하수.xlsx", header=1)
df23_agri = pd.read_excel("세부용도별_2023년_서울특별시_농·어업용 지하수.xlsx", header=1)
df23_ind  = pd.read_excel("세부용도별_2023년_서울특별시_공업용 지하수.xlsx", header=1)

# -----------------------------
# 2) 필요한 열만 선택(행정구역, 총계) + 총계 숫자 변환
#    만약 '총계' 대신 '합계'인 경우를 대비해 간단 처리
# -----------------------------
def pick_and_clean(df):
    # 열 이름 통일
    cols = df.columns
    if '총계' not in cols and '합계' in cols:
        df = df.rename(columns={'합계': '총계'})
    # 필요한 열만
    df = df[['행정구역', '총계']]
    # 숫자형 변환 (문자/콤마 등 있을 때 대비)
    df['총계'] = pd.to_numeric(df['총계'], errors='coerce')
    # 결측 제거(필요시)
    df = df.dropna(subset=['총계'])
    return df

dfs = [
    (2021, '생활용', pick_and_clean(df21_life)),
    (2021, '농업용', pick_and_clean(df21_agri)),
    (2021, '공업용', pick_and_clean(df21_ind)),
    (2022, '생활용', pick_and_clean(df22_life)),
    (2022, '농업용', pick_and_clean(df22_agri)),
    (2022, '공업용', pick_and_clean(df22_ind)),
    (2023, '생활용', pick_and_clean(df23_life)),
    (2023, '농업용', pick_and_clean(df23_agri)),
    (2023, '공업용', pick_and_clean(df23_ind)),
]

# -----------------------------
# 3) 연도×용도 합계 만들기 (아주 단순: '총계' 열의 합)
# -----------------------------
rows = []
for year, usage, df in dfs:
    total = df['총계'].sum()
    rows.append([year, usage, total])

df_year_usage = pd.DataFrame(rows, columns=['연도', '용도', '사용량'])
print(df_year_usage)

# -----------------------------
# 4) (선택) CSV 저장
# -----------------------------
df_year_usage.to_csv("서울_지하수_3개년_생활_농업_공업_통합집계.csv", index=False, encoding='utf-8-sig')

# -----------------------------
# 5) 시각화 1: 연도별-용도별 그룹 막대 (seaborn)
# -----------------------------
plt.figure(figsize=(8, 5))
sns.barplot(data=df_year_usage, x='연도', y='사용량', hue='용도')
plt.title('서울특별시 지하수 사용량 (생활용·농업용·공업용) - 3개년 그룹 막대')
plt.xlabel('연도')
plt.ylabel('사용량 합계')
plt.tight_layout()
plt.show()

# -----------------------------
# 6) 시각화 2: 연도별 누적 막대 (matplotlib)
# -----------------------------
# 피벗: 행=연도, 열=용도, 값=사용량
pivot = df_year_usage.pivot(index='연도', columns='용도', values='사용량').fillna(0)

# 연도 순서
years = sorted(pivot.index.tolist())
x = range(len(years))

# 누적용 바닥
bottom = [0] * len(years)

plt.figure(figsize=(8, 5))
for usage in ['생활용', '농업용', '공업용']:
    if usage in pivot.columns:
        y = [pivot.loc[y, usage] for y in years]
        plt.bar(x, y, bottom=bottom, label=usage)
        # 바닥 업데이트
        bottom = [b + v for b, v in zip(bottom, y)]

plt.xticks(list(x), [str(y) for y in years])
plt.title('서울특별시 지하수 사용량 (생활용·농업용·공업용) - 3개년 누적 막대')
plt.xlabel('연도')
plt.ylabel('사용량 합계')
plt.legend()
plt.tight_layout()
plt.show()
